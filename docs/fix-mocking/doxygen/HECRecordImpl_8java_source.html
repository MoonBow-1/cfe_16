135 @Override
136 public SyslogMessage toSyslogMessage() {
137 final long currentEpochMillis = Instant.now().toEpochMilli();
138 final SDElement structuredMetadata = structuredDataParams(currentEpochMillis);
139
140 final SyslogMessage syslogMessage;
141 if (this.time().parsed()) {
142
143 /*
144 * Creates a Syslogmessage with a time stamp
145 */
146 LOGGER.debug("Creating new syslog message with timestamp");
147 syslogMessage = new SyslogMessage()
148 .withTimestamp(this.time().instant(currentEpochMillis))
149 .withSeverity(this.severity)
150 .withAppName("capsulated")
151 .withHostname(this.hostName)
152 .withFacility(this.facility)
153 .withSDElement(structuredMetadata)
154 .withSDElement(this.headerInfo.asSDElement())
155 .withMsg(this.event().asString());
156
157 }
158 else {
159 /*
160 * Creates a Syslogmessage without timestamp, because the time is already given
161 * in the request.
162 */
163 LOGGER.debug("Creating new syslog message without timestamp");
164 syslogMessage = new SyslogMessage()
165 .withSeverity(this.severity)
166 .withAppName("capsulated")
167 .withHostname(this.hostName)
168 .withFacility(this.facility)
169 .withSDElement(structuredMetadata)
170 .withSDElement(this.headerInfo.asSDElement())
171 .withMsg(this.event().asString());
172 }
173
174 return syslogMessage;

228 /*
229 * Gets the data from the HTTP Event Data and adds it to SD Element as SD
230 * Parameters.
231 */
232 private SDElement structuredDataParams(final long fallbackEpoch) {
233 LOGGER.debug("Setting Structured Data params");
234 final SDElement metadataSDE = new SDElement("cfe_16-metadata@48577");
235
236 if (this.authenticationToken() != null) {
237 LOGGER.debug("Setting authentication token");
238 metadataSDE.addSDParam("authentication_token", this.authenticationToken());
239 }
240
241 if (this.channel() != null) {
242 LOGGER.debug("Setting channel");
243 metadataSDE.addSDParam("channel", this.channel());
244 }
245
246 if (this.ackID() != null) {
247 LOGGER.debug("Setting ack id");
248 metadataSDE.addSDParam("ack_id", String.valueOf(this.ackID()));
249 }
250
251 if (this.time().source() != null) {
252 LOGGER.debug("Setting time source");
253 metadataSDE.addSDParam("time_source", this.time().source());
254 }
255
256 LOGGER.debug("Setting time_parsed and time");
257 metadataSDE.addSDParam("time_parsed", String.valueOf(this.time().parsed()));
258 metadataSDE.addSDParam("time", String.valueOf(this.time().instant(fallbackEpoch)));
259 if (this.time().parsed()) {
260 LOGGER.debug("TimeParsed was true");
261 metadataSDE.addSDParam("generated", "false");
262 }
263 else {
264 LOGGER.debug("TimeParsed was false");
265 metadataSDE.addSDParam("generated", "true");
266 }
267
268 return metadataSDE;
269 }
