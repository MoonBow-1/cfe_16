140 public Response sendEvents(HttpServletRequest request, String channel, String eventInJson) {
141 LOGGER.debug("Sending events to channel <{}>", channel);
142 if (this.tokenManager.tokenIsMissing(request)) {
143 throw new AuthenticationTokenMissingException("Authentication token must be provided");
144 }
145 // AspectLoggerWrapper.logMetricCounter(null, "metric_counter", 10);

147 // MetricDurationOptionsImpl.MetricDuration.P10S);
148 String authHeader = request.getHeader("Authorization");
149 LOGGER.debug("Creating new Header Info");
150
151 final HeaderInfo headerInfo = this.headerInfo(request);
152
153 String authToken;
154 if (tokenManager.isTokenInBasic(authHeader)) {
155 LOGGER.debug("Token was provided via Basic");
156 authToken = this.tokenManager.getTokenFromBasic(authHeader);
157 }
158 else {
159 LOGGER.debug("Token was provided via header");
160 authToken = authHeader;
161 }
162
163 // if there is no channel, we'll use the default channel
164 if (channel == null) {
165 channel = Session.DEFAULT_CHANNEL;
166 LOGGER.debug("Channel was not provided, using <{}>", channel);
167 }
168
169 Session session = this.sessionManager.getOrCreateSession(authToken);
170
171 // if the channel is not in the session, let's add the channel into it
172 if (!session.doesChannelExist(channel)) {
173 LOGGER.debug("Adding channel <{}>", channel);
174 session.addChannel(channel);
175 }
176
177 acknowledgements.initializeContext(authToken, channel);
178 final int ackId = acknowledgements.getCurrentAckValue(authToken, channel);
179 final boolean incremented = acknowledgements.incrementAckValue(authToken, channel);
180 if (!incremented) {
181 throw new InternalServerErrorException("Ack value couldn't be incremented.");
182 }
183 final Ack ack = new Ack(ackId, false);
184 final boolean addedAck = acknowledgements.addAck(authToken, channel, ack);
185 if (!addedAck) {
186 throw new InternalServerErrorException("Ack ID " + ackId + " couldn't be added to the Ack set.");
187 }
188
189 Response responseToReturn;
190
191 try {
192 this.connection
193 .sendMessages(new SyslogBatch(new HECBatch(authToken, channel, eventInJson, headerInfo).toHECRecordList()).asSyslogMessages());
194
195 final boolean shouldAck = !channel.equals(Session.DEFAULT_CHANNEL);
196
197 if (shouldAck) {
198 final boolean acked = acknowledgements.acknowledge(authToken, channel, ackId);
199 if (!acked) {
200 throw new InternalServerErrorException("Ack ID " + ackId + " not Acked.");
201 }
202 else {
203 responseToReturn = new AcknowledgedJsonResponse("Success", ackId);
204 }
205 }
206 else {
207 responseToReturn = new JsonResponse("Success");
208 }
209 }
210 catch (final JsonSyntaxException | IOException e) {
211 final ExceptionEventContext exceptionEventContext = new ExceptionEventContext(
212 headerInfo,
213 request.getHeader("user-agent"),
214 request.getRequestURI(),
215 request.getRemoteHost()
216 );
217 final ExceptionEvent event = new ExceptionEvent(exceptionEventContext, UUID.randomUUID(), e);
218 event.logException();
219 responseToReturn = new ExceptionJsonResponse(event);
220 }
221
222 return responseToReturn;
223 }

229 public JsonNode getAcks(HttpServletRequest request, String channel, JsonNode requestedAcksInJson) {
230
231 // filter out error cases
232 // authentication header is required always
233 if (this.tokenManager.tokenIsMissing(request)) {
234 throw new AuthenticationTokenMissingException("Authentication token must be provided");
235 }
236
237 String authHeader = request.getHeader("Authorization");
238
239 String authToken;
240 if (tokenManager.isTokenInBasic(authHeader)) {
241 LOGGER.debug("Token was provided via Basic");
242 authToken = this.tokenManager.getTokenFromBasic(authHeader);
243 }
244 else {
245 LOGGER.debug("Token was provided via header");
246 authToken = authHeader;
247 }
248
249 // channel is required
250 if (channel == null) {
251 throw new ChannelNotProvidedException("Channel must be provided when requesting ack statuses");
252 }
253
254 // session is also required
255 Session session = this.sessionManager.getSession(authToken);
256 if (session == null) {
257 throw new SessionNotFoundException("Session not found for auth token " + authToken);
258 }
259
260 // if channel is not inside Session, it is considered an error case
261 if (!session.doesChannelExist(channel)) {
262 throw new ChannelNotFoundException();
263 }
264 session.touch();
265
266 ObjectNode responseNode = objectMapper.createObjectNode();
267 JsonNode requestedAckStatuses = this.acknowledgements
268 .getRequestedAckStatuses(authToken, channel, requestedAcksInJson);
269 responseNode.put("acks", requestedAckStatuses);
270 return responseNode;
