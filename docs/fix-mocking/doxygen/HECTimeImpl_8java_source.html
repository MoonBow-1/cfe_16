63 @Override
64 public long instant(final long defaultValue) {
65 final long returnedTime;
66 JsonNode timeNode;
67
68 try {
69 if (timeNode1.hasTime()) {
70 timeNode = timeNode1.asTimeJsonNode();
71 }
72 else {
73 timeNode = new TextNode("");
74 }
75 }
76 catch (final EventFieldException e) {
77 timeNode = new TextNode("");
78 }
79
80 // No time provided in the event
81 if (timeNode == null || timeNode.asText().isEmpty()) {
82 // Use default value
83 returnedTime = defaultValue;
84 }
85 // Check if time is a double and convert to long
86 else if (timeNode.isDouble()) {
87 returnedTime = this.removeDecimal(timeNode.asDouble());
88
89 }
90 // Time is a number, no calculations required
91 else if (timeNode.canConvertToLong()) {
92 returnedTime = timeNode.asLong();
93
94 }
95 // Time is a String
96 else if (timeNode.isTextual()) {
97 // Try to convert the String to a long (if not convertable, default to 0L)
98 final long tryAsLong = timeNode.asLong(0L);
99 if (tryAsLong != 0L) {
100 returnedTime = tryAsLong;
101 }
102 // No time found in current or previous event
103 else {
104 // Use default value
105 returnedTime = defaultValue;
106 }
107 }
108 // Unknown format
109 else {
110 // Use default value
111 returnedTime = defaultValue;
112 }
113
114 return returnedTime;
115 }

121 private long removeDecimal(final double doubleValue) {
