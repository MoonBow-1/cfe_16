131 @Override
132 public SyslogMessage toSyslogMessage() {
133 final long currentEpochMillis = Instant.now().toEpochMilli();
134 final SDElement structuredMetadata = structuredDataParams(currentEpochMillis);
135
136 final SyslogMessage syslogMessage;
137 if (this.time().parsed()) {
138
139 /*
140 * Creates a Syslogmessage with a time stamp
141 */
142 LOGGER.debug("Creating new syslog message with timestamp");
143 syslogMessage = new SyslogMessage()
144 .withTimestamp(this.time().asInstant(currentEpochMillis))
145 .withSeverity(this.severity)
146 .withAppName("capsulated")
147 .withHostname(this.hostName)
148 .withFacility(this.facility)
149 .withSDElement(structuredMetadata)
150 .withSDElement(this.headerInfo.asSDElement())
151 .withMsg(this.event().asString());
152
153 }
154 else {
155 /*
156 * Creates a Syslogmessage without timestamp, because the time is already given
157 * in the request.
158 */
159 LOGGER.debug("Creating new syslog message without timestamp");
160 syslogMessage = new SyslogMessage()
161 .withSeverity(this.severity)
162 .withAppName("capsulated")
163 .withHostname(this.hostName)
164 .withFacility(this.facility)
165 .withSDElement(structuredMetadata)
166 .withSDElement(this.headerInfo.asSDElement())
167 .withMsg(this.event().asString());
168 }
169
170 return syslogMessage;
171 }

219 /*
220 * Gets the data from the HTTP Event Data and adds it to SD Element as SD
221 * Parameters.
222 */
223 private SDElement structuredDataParams(final long fallbackEpoch) {
224 LOGGER.debug("Setting Structured Data params");
225 final SDElement metadataSDE = new SDElement("cfe_16-metadata@48577");
226
227 if (this.authenticationToken() != null) {
228 LOGGER.debug("Setting authentication token");
229 metadataSDE.addSDParam("authentication_token", this.authenticationToken());
230 }
231
232 if (this.channel() != null) {
233 LOGGER.debug("Setting channel");
234 metadataSDE.addSDParam("channel", this.channel());
235 }
236
237 if (this.time().source() != null) {
238 LOGGER.debug("Setting time source");
239 metadataSDE.addSDParam("time_source", this.time().source());
240 }
241
242 LOGGER.debug("Setting time_parsed and time");
243 metadataSDE.addSDParam("time_parsed", String.valueOf(this.time().parsed()));
244 metadataSDE.addSDParam("time", String.valueOf(this.time().asInstant(fallbackEpoch)));
245 if (this.time().parsed()) {
246 LOGGER.debug("TimeParsed was true");
247 metadataSDE.addSDParam("generated", "false");
248 }
249 else {
250 LOGGER.debug("TimeParsed was false");
251 metadataSDE.addSDParam("generated", "true");
252 }
253
254 return metadataSDE;
255 }
