60 @Override
61 public EventMessage asEventMessage() throws EventFieldException {
62 final EventMessage eventMessage;
63 // Event field completely missing
64 if (!this.asPayloadJsonNode().has("event")) {
65 throw new EventFieldException("Event field is missing");
66 }
67 // Event field contains subfield "message"
68 else if (
69 this.asPayloadJsonNode().get("event").isObject() && this.asPayloadJsonNode().get("event").has("message")
70 ) {
71 if (
72 this.asPayloadJsonNode().get("event").get("message").isTextual()
73 && !Objects.equals(this.asPayloadJsonNode().get("event").get("message").asText(), "")
74 ) {
75 eventMessage = new EventMessageImpl(this.asPayloadJsonNode().get("event").get("message").asText());
76 }
77 else {
78 throw new EventFieldException("Event field was not textual");
79 }
80 }
81 // Event field has a String value
82 else if (
83 this.asPayloadJsonNode().get("event").isTextual()
84 && !Objects.equals(this.asPayloadJsonNode().get("event").asText(), "")
85 ) {
86 eventMessage = new EventMessageImpl(this.jsonNode.get("event").asText());
87 }
88 else {
89 throw new EventFieldException("Event field was not textual");
90 }
91
92 return eventMessage;
93 }
94
95 @Override
96 public JsonNode asPayloadJsonNode() {
97 if (this.jsonNode != null && this.jsonNode.isObject()) {
98 return this.jsonNode;
99 }
100 throw new IllegalArgumentException("jsonEvent node not valid");
101 }
